–î–ª—è –≤–∏–∑–Ω–∞—á–µ–Ω–Ω—è —Ç–æ–≥–æ, —á–∏ —î —Ä—É—Ö –ø–µ—Ä—ñ–æ–¥–∏—á–Ω–∏–º –∞–±–æ —Ö–∞–æ—Ç–∏—á–Ω–∏–º, –æ—Å–æ–±–ª–∏–≤–æ –¥–ª—è –≤–µ–ª–∏–∫–∏—Ö \(t\), —Ä–µ–∫–æ–º–µ–Ω–¥—É—î—Ç—å—Å—è —Ä–æ–∑–≥–ª—è–¥–∞—Ç–∏ –≤–Ω–µ—Å–æ–∫ —Å–∏—Å—Ç–µ–º–∏ –≤ –Ω–∞–π–±—ñ–ª—å—à–∏–π –ø–æ–∫–∞–∑–Ω–∏–∫ –õ—è–ø—É–Ω–æ–≤–∞ (–ù–ü–õ), –æ—Å–∫—ñ–ª—å–∫–∏ –¥—ñ–∞–º–µ—Ç—Ä \(n\)-—Ä–æ–∑–º—ñ—Ä–Ω–æ–≥–æ –µ–ª—ñ–ø—Å–æ—ó–¥–∞ –ø–æ—á–∏–Ω–∞—î –∑–∞–ª–µ–∂–∞—Ç–∏ –≤—ñ–¥ –Ω—å–æ–≥–æ  [2]. –°–∞–º–µ –ù–ü–õ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î—Ç—å—Å—è –¥–ª—è –∫—ñ–ª—å–∫—ñ—Å–Ω–æ—ó –æ—Ü—ñ–Ω–∫–∏ –ø–µ—Ä–µ–¥–±–∞—á—É–≤–∞–Ω–æ—Å—Ç—ñ —Å–∏—Å—Ç–µ–º, –æ—Å–∫—ñ–ª—å–∫–∏ –µ–∫—Å–ø–æ–Ω–µ–Ω—Ü—ñ–∞–ª—å–Ω–∞ —Ä–æ–∑–±—ñ–∂–Ω—ñ—Å—Ç—å –æ–∑–Ω–∞—á–∞—î, —â–æ –≤ —Å–∏—Å—Ç–µ–º—ñ, –¥–µ –ø–æ—á–∞—Ç–∫–æ–≤–µ –∑–±—É—Ä–µ–Ω–Ω—è –±—É–ª–æ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ –º–∞–ª–∏–º, –ø–æ—á–∏–Ω–∞—é—Ç—å—Å—è –≤—Ç—Ä–∞—Ç–∏ –ø–µ—Ä–µ–¥–±–∞—á—É–≤–∞–Ω–Ω–æ—Å—Ç—ñ. –û–¥–Ω–∞–∫ —Å–ª—ñ–¥ –∑–∞–∑–Ω–∞—á–∏—Ç–∏, —â–æ —ñ–Ω—à—ñ –ø–æ–∫–∞–∑–Ω–∏–∫–∏ —Ç–∞–∫–æ–∂ –º—ñ—Å—Ç—è—Ç—å –≤–∞–∂–ª–∏–≤—É —ñ–Ω—Ñ–æ—Ä–º–∞—Ü—ñ—é —â–æ–¥–æ —Å—Ç—ñ–π–∫–æ—Å—Ç—ñ —Å–∏—Å—Ç–µ–º–∏, –≤ —Ç–æ–º—É —á–∏—Å–ª—ñ –ø—Ä–æ –Ω–∞–ø—Ä—è–º–∫–∏ –∑–±—ñ–∂–Ω–æ—Å—Ç—ñ —Ç–∞ —Ä–æ–∑–±—ñ–∂–Ω–æ—Å—Ç—ñ —Ç—Ä–∞—î–∫—Ç–æ—Ä—ñ–π  [4].

–Ü—Å–Ω—É–≤–∞–Ω–Ω—è –ø—Ä–∏–Ω–∞–π–º–Ω—ñ –æ–¥–Ω–æ–≥–æ –ø–æ–∑–∏—Ç–∏–≤–Ω–æ–≥–æ –ü–õ –∑–∞–∑–≤–∏—á–∞–π —Ä–æ–∑–≥–ª—è–¥–∞—î—Ç—å—Å—è —è–∫ —Å–∏–ª—å–Ω–∏–π —ñ–Ω–¥–∏–∫–∞—Ç–æ—Ä —Ö–∞–æ—Å—É. –ü–æ–∑–∏—Ç–∏–≤–Ω–∏–π –ü–õ –æ–∑–Ω–∞—á–∞—î, —â–æ –ø–æ—á–∞—Ç–∫–æ–≤–æ –±–ª–∏–∑—å–∫—ñ —Ç—Ä–∞—î–∫—Ç–æ—Ä—ñ—ó —É —Ñ–∞–∑–æ–≤–æ–º—É –ø—Ä–æ—Å—Ç–æ—Ä—ñ, —á—É—Ç–ª–∏–≤—ñ –¥–æ –ø–æ—á–∞—Ç–∫–æ–≤–∏—Ö —É–º–æ–≤ —ñ —Ä–æ–∑—Ö–æ–¥—è—Ç—å—Å—è –µ–∫—Å–ø–æ–Ω–µ–Ω—Ü—ñ–∞–ª—å–Ω–æ —à–≤–∏–¥–∫–æ. –ù–µ–≥–∞—Ç–∏–≤–Ω–∏–π –ü–õ –≤—ñ–¥–ø–æ–≤—ñ–¥–∞—î –≤–∏–ø–∞–¥–∫–∞–º, –∫–æ–ª–∏ —Ç—Ä–∞—î–∫—Ç–æ—Ä—ñ—ó –∑–∞–ª–∏—à–∞—é—Ç—å—Å—è –±–ª–∏–∑—å–∫–∏–º–∏ –æ–¥–Ω–∞ –¥–æ –æ–¥–Ω–æ—ó, –∞–ª–µ —Ü–µ –Ω–µ –æ–±–æ–≤‚Äô—è–∑–∫–æ–≤–æ –æ–∑–Ω–∞—á–∞—î —Å—Ç–∞–±—ñ–ª—å–Ω—ñ—Å—Ç—å, —ñ –º–∏ –ø–æ–≤–∏–Ω–Ω—ñ –¥–æ—Å–ª—ñ–¥–∏—Ç–∏ –Ω–∞—à—É —Å–∏—Å—Ç–µ–º—É –±—ñ–ª—å—à –¥–µ—Ç–∞–ª—å–Ω–æ. –ù—É–ª—å–æ–≤—ñ –∞–±–æ –¥—É–∂–µ –±–ª–∏–∑—å–∫—ñ –¥–æ –Ω—É–ª—è –ø–æ–∫–∞–∑–Ω–∏–∫–∏ –≤–∫–∞–∑—É—é—Ç—å –Ω–∞ —Ç–µ, —â–æ –∑–±—É—Ä–µ–Ω–Ω—è, –ø—Ä–∞–∫—Ç–∏—á–Ω–æ –Ω–µ –≤–ø–ª–∏–≤–∞—é—Ç—å –Ω–∞ –µ–≤–æ–ª—é—Ü—ñ—é —Ç—Ä–∞—î–∫—Ç–æ—Ä—ñ–π –¥–∏–Ω–∞–º—ñ—á–Ω–æ—ó —Å–∏—Å—Ç–µ–º–∏.

–£ –∑–≤‚Äô—è–∑–∫—É –∑ –≤–µ–ª–∏–∫–æ—é –∑–∞—Ü—ñ–∫–∞–≤–ª–µ–Ω—ñ—Å—Ç—é –≤ –ü–õ, –∑‚Äô—è–≤–ª—è—î—Ç—å—Å—è –≤—Å–µ –±—ñ–ª—å—à–µ —ñ–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç—ñ–≤ –¥–ª—è —Ä–æ–∑—Ä–∞—Ö—É–Ω–∫—É. –ù–∞ –∂–∞–ª—å, –¥–æ—Å—ñ –Ω–µ –æ—Ç—Ä–∏–º–∞–Ω–æ –∑–∞–≥–∞–ª—å–Ω–æ–ø—Ä–∏–π–Ω—è—Ç–æ–≥–æ —Ç–∞ —É–Ω—ñ–≤–µ—Ä—Å–∞–ª—å–Ω–æ–≥–æ –º–µ—Ç–æ–¥—É –æ—Ü—ñ–Ω–∫–∏ –≤—Å—å–æ–≥–æ —Å–ø–µ–∫—Ç—Ä—É –ø–æ–∫–∞–∑–Ω–∏–∫—ñ–≤ –õ—è–ø—É–Ω–æ–≤–∞ –∑–∞ –∑–Ω–∞—á–µ–Ω–Ω—è–º–∏ —á–∞—Å–æ–≤–æ–≥–æ —Ä—è–¥—É. –û–¥–Ω—ñ –∑ –Ω–∞–π–±—ñ–ª—å—à –ø–æ—à–∏—Ä–µ–Ω–∏—Ö —ñ –ø–æ–ø—É–ª—è—Ä–Ω–∏—Ö –∞–ª–≥–æ—Ä–∏—Ç–º—ñ–≤ –±—É–ª–∏ –∑–∞—Å—Ç–æ—Å–æ–≤–∞–Ω—ñ –í–æ–ª—å—Ñ–æ–º —Ç–∞ —ñ–Ω.  [5], –°–∞–Ω–æ —ñ –°–∞–≤–∞–¥–æ—é  [6], –∞ –ø—ñ–∑–Ω—ñ—à–µ –≤–¥–æ—Å–∫–æ–Ω–∞–ª–µ–Ω—ñ –ï–∫–º–∞–Ω–æ–º  [7], –†–æ–∑–µ–Ω—à—Ç–µ–π–Ω–æ–º  [8], –ü–∞—Ä–ª—ñ—Ü–æ–º  [9], –ë–∞–ª—å—Ü–µ—Ä–∂–∞–∫–æ–º —Ç–æ—â–æ  [10].

9.1.1.1 –ú–µ—Ç–æ–¥ –ï–∫–º–∞–Ω–∞

–ü–æ-–ø–µ—Ä—à–µ, –∑–≥—ñ–¥–Ω–æ –∑ –ø—ñ–¥—Ö–æ–¥–æ–º –ï–∫–º–∞–Ω–∞ —Ç–∞ —ñ–Ω.  [7], –º–∏ –ø–æ–≤–∏–Ω–Ω—ñ —Ä–µ–∫–æ–Ω—Å—Ç—Ä—É—é–≤–∞—Ç–∏ –¥–∏–Ω–∞–º—ñ–∫—É –∞—Ç—Ä–∞–∫—Ç–æ—Ä–∞ –∑ —á–∞—Å–æ–≤–æ–≥–æ —Ä—è–¥—É \(\{x(i) \,|\, i=1,...N \}\) –∑ —Ä–æ–∑–º—ñ—Ä–Ω—ñ—Å—Ç—é –≤–∫–ª–∞–¥–µ–Ω—å \(d_E\), —ñ –ø—ñ—Å–ª—è —Ü—å–æ–≥–æ –ø–æ–±—É–¥—É–≤–∞—Ç–∏ \(d_E\)-–≤–∏–º—ñ—Ä–Ω—É –æ—Ä–±—ñ—Ç—É, —â–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è—î —á–∞—Å–æ–≤—É –µ–≤–æ–ª—é—Ü—ñ—é

\[ \vec{X}(i) = \left[ x(i), x(i+1), ..., x(i+(d_E - 1)) \right], \; \text{–¥–ª—è} \; i=1,...,N-d_E+1. \]

–î–∞–ª—ñ, –º–∏ –º–∞—î–º–æ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ –Ω–∞–π–±–ª–∏–∂—á—ñ –¥–æ \(\vec{X}(i)\) —Ç—Ä–∞—î–∫—Ç–æ—Ä—ñ—ó:

\[ \| \vec{X}(i) - \vec{X}(j) \| = \max_{0\leq \alpha \leq d_E-1} \left| x(i+\alpha) - x(j+\alpha) \right|. \tag{9.3}\]

–°–æ—Ä—Ç—É—î–º–æ \(x(i)\) —Ç–∞–∫, —â–æ–± \(x(\Pi(1)) \leq x(\Pi(2)) \leq ... \leq x(\Pi(N))\) —ñ –∑–±–µ—Ä—ñ–≥–∞—î–º–æ –ø–µ—Ä–µ—Å—Ç–∞–Ω–æ–≤–∫—É \(\Pi\) —Ç–∞ —ó—ó –∑–≤–æ—Ä–æ—Ç–Ω—é –≤–µ—Ä—Å—ñ—é \(\Pi^-1\). –î–∞–ª—ñ –Ω–∞–º–∞–≥–∞—î–º–æ—Å—å –∑–Ω–∞–π—Ç–∏ —Å—É—Å—ñ–¥—ñ–≤ \(x(i)\), –ø–µ—Ä–µ–≥–ª—è–¥–∞—é—á–∏ \(k=\Pi^{-1}(i)\) —ñ —Å–∫–∞–Ω—É—î–º–æ \(x(\Pi(s))\) –ø—Ä–∏ \(s=k+1, k+2,...\) —ñ \(k-1, k-2,...\) –¥–æ —Ç–∏—Ö –ø—ñ—Ä, –¥–æ –ø–æ–∫–∏ –Ω–µ –≤–∏–∫–æ–Ω–∞—î—Ç—å—Å—è —É–º–æ–≤–∞ \(x(\Pi(ùë†)) - x(i) > r\). –î–ª—è –≤–∏–±—Ä–∞–Ω–æ—ó —Ä–æ–∑–º—ñ—Ä–Ω–æ—Å—Ç—ñ –≤–∫–ª–∞–¥–µ–Ω–Ω—è \(d_E > 1\) –≤–∏–±–∏—Ä–∞—î–º–æ –≤–∏–±–∏—Ä–∞—î–º–æ –∑–Ω–∞—á–µ–Ω–Ω—è \(s\) –∑–∞ —É–º–æ–≤–∏

\[ \left| x(\Pi(s) + \alpha) - x(j + \alpha) \right| \leq r, \; \text{–¥–ª—è} \; \alpha=0,1,...,d_E-1. \]

–ü—ñ—Å–ª—è —Ä–µ–∫–æ–Ω—Å—Ç—Ä—É–∫—Ü—ñ—ó —Å–∏—Å—Ç–µ–º –¥–æ —Ä–æ–∑–º—ñ—Ä–Ω–æ—Å—Ç—ñ \(d_E\) –ø–æ—Ç—Ä—ñ–±–Ω–æ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ –º–∞—Ç—Ä–∏—Ü—é \(M_i\) —Ä–æ–∑–º—ñ—Ä–Ω–æ—Å—Ç—ñ \(d_E \times d_E\), —è–∫–∞ –æ–ø–∏—Å—É–≤–∞—Ç–∏–º–µ —á–∞—Å–æ–≤—É –µ–≤–æ–ª—é—Ü—ñ—é –≤–µ–∫—Ç–æ—Ä—ñ–≤, —ñ–∑ –æ—Ç–æ—á–µ–Ω–Ω—è —Ç—Ä–∞—î–∫—Ç–æ—Ä—ñ—ó \(\vec{X}(i)\), —ñ —Ç–µ, —è–∫ –≤–æ–Ω–∏ –≤—ñ–¥–æ–±—Ä–∞–∂–∞—é—Ç—å—Å—è –Ω–∞ —Å—Ç–∞–Ω \(\vec{X}(i+1)\). –ú–∞—Ç—Ä–∏—Ü—è \(M_i\) –æ—Ç—Ä–∏–º—É—î—Ç—å—Å—è —à–ª—è—Ö–æ–º –ø–æ—à—É–∫—É —Å—É—Å—ñ–¥—ñ–≤

\[ M_i(\vec{X}(i) - \vec{X}(j)) \approx \vec{X}(i+1) - \vec{X}(j+1). \tag{9.4}\]

–í–µ–∫—Ç–æ—Ä–∏ \(\vec{X}(i)-\vec{X}(j)\) –º–æ–∂—É—Ç—å —ñ –Ω–µ –ø–æ–∫—Ä–∏–≤–∞—Ç–∏ \(\mathbb{R}^{d_E}\). –£ —Ü—å–æ–º—É –≤–∏–ø–∞–¥–∫—É —Ç–∞–∫–∞ –Ω–µ–≤–∏–∑–Ω–∞—á–µ–Ω—ñ—Å—Ç—å –º–æ–∂–µ –ø—Ä–∏–∑–≤–µ—Å—Ç–∏ –¥–æ —Ö–∏–±–Ω–∏—Ö –ø–æ–∫–∞–∑–Ω–∏–∫—ñ–≤, —è–∫—ñ –º–æ–∂—É—Ç—å –∑—ñ–ø—Å—É–≤–∞—Ç–∏ –∞–Ω–∞–ª—ñ–∑. –î–ª—è –ø–æ–¥–æ–ª–∞–Ω–Ω—è —Ç–∞–∫–∏—Ö –ø–µ—Ä–µ—à–∫–æ–¥ –ø—Ä–æ–µ–∫—Ü—ñ—è —Ç—Ä–∞—î–∫—Ç–æ—Ä—ñ–π –≤–∏–∑–Ω–∞—á–∞—î—Ç—å—Å—è –Ω–∞ –ø—ñ–¥–ø—Ä–æ—Å—Ç–æ—Ä—ñ —Ä–æ–∑–º—ñ—Ä–Ω–æ—Å—Ç—ñ \(d_M \leq d_E\). –¢–∞–∫–∏–º —á–∏–Ω–æ–º, –ø—Ä–æ—Å—Ç—ñ—Ä, –Ω–∞ —è–∫–æ–º—É –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è –¥–∏–Ω–∞–º—ñ–∫–∞, –≤—ñ–¥–ø–æ–≤—ñ–¥–∞—î –ª–æ–∫–∞–ª—å–Ω—ñ–π —Ä–æ–∑–º—ñ—Ä–Ω–æ—Å—Ç—ñ \(d_M\), –∞ \(d_E\) –º–∞—î –±—É—Ç–∏ –¥–µ—â–æ –±—ñ–ª—å—à–∏–º –∑–∞ \(d_M\), —â–æ–± —É–Ω–∏–∫–Ω—É—Ç–∏ –Ω–∞—è–≤–Ω–æ—Å—Ç—ñ —Ö–∏–±–Ω–∏—Ö —Å—É—Å—ñ–¥—ñ–≤  [11,12]. –ó–≤—ñ–¥—Å–∏ –≤–∏–ø–ª–∏–≤–∞—î, —â–æ —Ç—Ä–∞—î–∫—Ç–æ—Ä—ñ—è \(\vec{X}(i)\) –∞—Å–æ—Ü—ñ—é—î—Ç—å—Å—è –∑ \(d_M\)-–≤–∏–º—ñ—Ä–Ω–∏–º –≤–µ–∫—Ç–æ—Ä–æ–º

\[ \vec{X}(i) = \left[ x(i), x(i+\tau), ..., x(i+(d_M - 1)\tau) \right] = \left[ x(i), x(i+\tau), ..., x(i+d_E - 1) \right], \tag{9.5}\]

–¥–µ \(\tau=(d_E-1)/(d_M-1)\). –ö–æ–ª–∏ \(\tau>1\), —É–º–æ–≤–∞ (9.4) –∑–∞–º—ñ–Ω—é—î—Ç—å—Å—è –Ω–∞—Å—Ç—É–ø–Ω–∏–º –≤–∏—Ä–∞–∑–æ–º:

\[ M_i(\vec{X}(i) - \vec{X}(j)) \approx \vec{X}(i+\tau) - \vec{X}(j+\tau). \tag{9.6}\]

–ú–∞—Ç—Ä–∏—Ü—è \(M_i\) –≤–∏–∑–Ω–∞—á–∞—î—Ç—å—Å—è –º–µ—Ç–æ–¥–æ–º –Ω–∞–π–º–µ–Ω—à–∏—Ö –∫–≤–∞–¥—Ä–∞—Ç—ñ–≤. –û—Å—Ç–∞–Ω–Ω—ñ–º –∫—Ä–æ–∫–æ–º —î QR –¥–µ–∫–æ–º–ø–æ–∑–∏—Ü—ñ—è –¥–ª—è –∑–Ω–∞—Ö–æ–¥–∂–µ–Ω–Ω—è –æ—Ä—Ç–æ–≥–æ–Ω–∞–ª—å–Ω–∏—Ö –º–∞—Ç—Ä–∏—Ü—å \(Q_i\) —ñ –≤–µ—Ä—Ö–Ω—ñ—Ö —Ç—Ä–∏–∫—É—Ç–Ω–∏—Ö –º–∞—Ç—Ä–∏—Ü—å \(R_i\) –ø—Ä–∏ —è–∫–∏—Ö

\[ M_{1+i\tau}Q_i = Q_{i+1}R_{i+1}, \; \text{–¥–ª—è} \; i=0,1,2,... . \]

–Ø–∫ –±—É–ª–æ –∑–∞–ø—Ä–æ–ø–æ–Ω–æ–≤–∞–Ω–æ –ï–∫–º–∞–Ω–æ–º  [7,13], –∑–Ω–∞—é—á–∏ \(K\) –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Ç–æ—á–æ–∫ –Ω–∞ –∞—Ç—Ä–∞–∫—Ç–æ—Ä—ñ, –¥—ñ–∞–≥–æ–Ω–∞–ª—å–Ω—ñ –≤–ª–∞—Å–Ω—ñ –∑–Ω–∞—á–µ–Ω–Ω—è –º–∞—Ç—Ä–∏—Ü—ñ \(R_i\) —Ç–∞ –∫—Ä–æ–∫ –¥–∏—Å–∫—Ä–µ—Ç–∏–∑–∞—Ü—ñ—ó \(\Delta t\), –º–æ–∂–Ω–∞ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ –Ω–∞—Å—Ç—É–ø–Ω–µ —Ä—ñ–≤–Ω—è–Ω–Ω—è –¥–ª—è –∑–Ω–∞—Ö–æ–¥–∂–µ–Ω–Ω—è \(k\)-–≥–æ –ü–õ:

\[ \lambda_k = \frac{1}{\Delta t}\frac{1}{\tau}\frac{1}{K}\sum_{i=0}^{K-1}\ln{(R_i)_{kk}}. \]

    if ret_type == 1:       # –ó–≤–∞–∂–∞—é—á–∏ –Ω–∞ –≤–∏–¥ —Ä—è–¥—É, –≤–∏–∫–æ–Ω—É—î–º–æ

    elif ret_type == 2:
        for_rec = for_rec.diff()
    elif ret_type == 3:
        for_rec = for_rec.pct_change()
    elif ret_type == 4:
        for_rec = for_rec.pct_change()
        for_rec -= for_rec.mean()
        for_rec /= for_rec.std()
    elif ret_type == 5: 
        for_rec = for_rec.pct_change()
        for_rec -= for_rec.mean()
        for_rec /= for_rec.std()
        for_rec = for_rec.abs()
    elif ret_type == 6:
        for_rec -= for_rec.mean()
        for_rec /= for_rec.std()

    for_rec = for_rec.dropna().values

    return for_rec
